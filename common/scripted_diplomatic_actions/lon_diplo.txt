scripted_diplomatic_actions = {
	join_lon = {
		icon = 1
		
		visible = {
			check_variable = { global.lon_leader = THIS }
			NOT = {
				is_in_array = { global.lon_members = ROOT }
			}
		}
		selectable = {
			ROOT = {
				has_war = no
			}
		}
		can_be_sent = {
			ROOT = {
				has_war = no
			}
		}

		requires_acceptance = yes
		show_acceptance_on_action_button = yes

		send_description = JOIN_LON_SEND_DESC
		receive_description = JOIN_LON_RECEIVE_DESC
		reject_title = JOIN_LON_REJECT_TITLE
		reject_description = JOIN_LON_REJECT_DESC

		complete_effect = {
			add_to_array = { global.lon_members = ROOT }
			news_event = lon.3
		}

		ai_acceptance = {
			base_reluctance = {
				base = -10
			}
			government_and_ideology = {
				base = 0
				modifier = {
					add = 5
					is_enemy_ideology = no
				}
			}
			aggression = {
				base = 0
				modifier = {
					add = 5
					has_war = no
				}
				modifier = {
					set_temp_variable = { threat_scale = lon_threat }
					multiply_temp_variable = { threat_scale = -10 }
					add = threat_scale
				}
			}
			opinion = {
				base = 0
				modifier = {
					set_temp_variable = { my_opinion = 0 }
					set_temp_variable = { my_opinion = opinion@ROOT }
					multiply_temp_variable = { my_opinion = 0.1 }
					add = my_opinion
				}
			}
		}

		ai_desire = {
			base = -5
			modifier = {
				add = 7.5
				FROM = { has_war = no }
			}
			modifier = {
				add = 7.5
				is_enemy_ideology = no
			}
			modifier = {
				add = 7.5
				FROM = { has_government = ROOT }
			}
			modifier = {
				add = 15
				if = {
					limit = {
						is_subject = yes
					}
					is_in_array = { global.lon_members = overlord }
				}
				else = {
					always = no
				}
			}
			modifier = {
				set_temp_variable = { threat_scale = FROM.lon_threat }
				add_to_temp_variable = { threat_scale = global.threat }
				add_to_temp_variable = { threat_scale = ROOT.lon_threat }
				multiply_temp_variable = { threat_scale = -7.5 }
				divide_temp_variable = { threat_scale = 3 }
				add_to_temp_variable = { threat_scale = 7.5 }
				add = threat_scale
			}
		}
	}
}
